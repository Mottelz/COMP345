597 void MapLoader::readMapFile(string &mapFile, Map &mapObject) throw(string)
598 {

600 
601  if (input.fail()) // Unable to open the file; may not exist or wrong file name or wrong directory

605 
606  return;
607  }
608 
609  string skip;
610  getline(input, skip, ']'); // Read until the first ] is encountered (after [Map])
611  getline(input, skip, ']'); // Read until the second ] is encountered (after [Continents])
612  getline(input, skip); // Read and discard the end of line
613 
614  string read;
615 
616  string continent;
617  vector<string> continents;
618  string continentName;
619  int continentBonus;
620  bool endOfContinentSection = false;
621 
622  while (!endOfContinentSection)
623  {
624  getline(input, read); // Read a line at a time
625  if (!read.empty())
626  {
627  stringstream line(read);
628  getline(line, continentName, '=');
629  getline(line, continent, '=');
630  continentBonus = atoi(continent.c_str());
631 
632  mapObject.insertContinent(continentName, continentBonus);
633  }
634  else // If the line is empty
635  {
636  while (getline(input, skip) && (skip.empty())); // Check if next line is also empty; skip until the next line is not empty
637  endOfContinentSection = true; // Exit loop because we've reached the [Territories] section (assuming a valid map despite the number of empty lines between sections)
638  }

640  }

642  vector<string> neighbouringTerritories;
643  string neighbouringTerritory;
644  string territoryName;
645  string coordinateX;

649  while(!input.eof()) // Read until the end of file

651  getline(input, read); // Read a line at a time
652  if (!read.empty())
653  {
654  stringstream line(read);
655  getline(line, territoryName, ',');
656  getline(line, coordinateX, ',');
