813 void MapLoader::readMapFile(string &mapFile, Map &mapObject) throw(string)
814 {
815  ifstream input(mapFile); // Open the file
816 
817  if (input.fail()) // Unable to open the file; may not exist or wrong file name or wrong directory
818  {
819  mapObject.setMapValidate(false);
820  throw string("ERROR: Unable to open the file. ");
821 
822  return;
823  }
824 
825  string skip;
826  getline(input, skip, ']'); // Read until the first ] is encountered (after [Map])
827  getline(input, skip, ']'); // Read until the second ] is encountered (after [Continents])
828  getline(input, skip); // Read and discard the end of line
829 
830  string read;
831 
832  string continent;
833  vector<string> continents;
834  string continentName;
835  int continentBonus;
836  bool endOfContinentSection = false;
837 
838  while (!endOfContinentSection)
839  {
840  getline(input, read); // Read a line at a time
841  if (!read.empty())
842  {
843  stringstream line(read);
844  getline(line, continentName, '=');
845  getline(line, continent, '=');
846  continentBonus = atoi(continent.c_str());
847 
848  mapObject.insertContinent(continentName, continentBonus);
849  }
850  else // If the line is empty
851  {
852  while (getline(input, skip) && (skip.empty())); // Check if next line is also empty; skip until the next line is not empty
853  endOfContinentSection = true; // Exit loop because we've reached the [Territories] section (assuming a valid map despite the number of empty lines between sections)
854  }
855 
856  }
857 
858  vector<string> neighbouringTerritories;
859  string neighbouringTerritory;
860  string territoryName;
861  string coordinateX;
862  string coordinateY;
863  float position[2];
864 
865  while(!input.eof()) // Read until the end of file
866  {
867  getline(input, read); // Read a line at a time
868  if (!read.empty())
869  {
870  stringstream line(read);
871  getline(line, territoryName, ',');
872  getline(line, coordinateX, ',');
873  getline(line, coordinateY, ',');
874  getline(line, continentName, ',');
875  while (line.good()) // Splitting every line into strings stored in the elements vector
876  {
877  getline(line, neighbouringTerritory, ',');
878  neighbouringTerritories.push_back(neighbouringTerritory);
879  }
880 
881  position[0] = stof(coordinateX);
882  position[1] = stof(coordinateY);
883  mapObject.insertTerritory(territoryName, position, continentName, neighbouringTerritories); // Add a territory for every line read
884  neighbouringTerritories.clear(); // Clear the vector for new iteration
885  }
886  }
887 
888  input.close(); // We're done reading the file
889 
890  mapObject.linkAllAdjacentTerritories();
891 
892  mapObject.linkAllTerritories();
893 
894  mapObject.isBadMap();
895 
896  cout << "Map File: " << mapObject.getMapName() << " is loaded successfully. " << endl;
897  cout << endl;
898 }
